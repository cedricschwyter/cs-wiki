Eulerian Circuit/Tour

A Eulerian circuit/tour is a Eulerian walk where there is an edge between the starting and the ending vertex. A Eulerian circuit in an undirected graph exists if and only if the graph is connected and all vertices have an even degree. def dfs(s, graph):
    q = Stack(s)
    dist = [None] * len(graph.nodes)
    enter = [None] * len(graph.nodes)
    leave = [None] * len(graph.nodes)
    dist[s] = 0
    enter[s] = 0
    t = 1
    while !q.isEmpty():
        u = q.pop()
        leave[u] = t
        t = t + 1
        for v in u.children:
            if !enter[v]:
                q.push(v)
                dist[v] = dist[u] + 1
                enter[v] = t
                t = t + 1
    return dist
